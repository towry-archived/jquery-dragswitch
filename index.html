source 
	$('#list1, #list2').dragswitch(function (d) {
 		d('.item').dragStart(function () {
 			console.log('drag start');
 		}).dragEnd(function () {
 			console.log('drag end');
 		}).config({
 			between: true
 		}).placeholder(function (p) {
 			p.css('border-color', 'green');
 		})
 	})
